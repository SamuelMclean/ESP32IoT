#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <string>
#include <WiFi.h>
#include <WiFiUdp.h>
#include <NTPClient.h>

const char* ssid     = "TelstraE9B3F0";
const char* password = "tkkpjaasphcen4u5";

const char* daysOfTheWeek[] = {"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};

// Define NTP Client to get time
WiFiUDP ntpUDP;
NTPClient timeClient(ntpUDP, "pool.ntp.org", 0, 60000);

// Set the LCD address to 0x27 for a 16 chars and 2 line display
LiquidCrystal_I2C lcd(0x27, 16, 2);

void setup() {
  // Initialize the LCD
  lcd.init();
  // Turn on the backlight
  lcd.backlight();

  // Connect to Wi-Fi
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Connecting to");
  lcd.setCursor(0, 1);
  lcd.print(String(ssid));

  Serial.print("Connecting to");
  Serial.println(ssid);
  WiFi.begin(ssid, password);
  int connection_attempts = 0;
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.print(".");
    if (connection_attempts == 10){
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Could not join");
      lcd.setCursor(0, 1);
      lcd.print(String(ssid));
      esp_deep_sleep_start();
    }
    
  }
  
  Serial.println("");
  Serial.println("WiFi connected.");
  
  // Initialize a NTPClient to get time
  timeClient.begin();
  timeClient.setTimeOffset(36000);
}

void loop() {
  timeClient.update();

  String time = timeClient.getFormattedTime();
  String day = daysOfTheWeek[timeClient.getDay()];
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print(day);
  lcd.setCursor(0, 1);
  lcd.print(time);

  int connection_attempts = 0;
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
    if (connection_attempts == 10){
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Could not join");
      lcd.setCursor(0, 1);
      lcd.print(String(ssid));
      esp_deep_sleep_start();
    }
    
  }

  delay(1000);
}
